4、定义变量，变量未赋值时，编译器会赋默认值。

package main
import "fmt"

// 声明全局变量法一
var (
	n3 = 300
	n4 = "hi"
)
func main()  {
	var n1, n2 int // 有默认值，多变量赋值
	s1 := "hello" // 定义变量,自动类型推导
	fmt.Println(n1, n2) // 0
	fmt.Println("s1=", s1)
	fmt.Println(n3, n4)
}
