工具：VS2019、OpenCV4.0.0

测试结果： 测试代码： #include <opencv.hpp> #include using namespace std;
using namespace cv;int main(int argc, char* argv[])
{
//初始化为500*500，每个像素为1个8bits
Mat grayim(500, 500, CV_8UC1);
//遍历Mat
for (size_t i = 0; i < grayim.rows; i++)
{
for (size_t j = 0; j < grayim.cols; j++)
{
//对像素进行赋值
*grayim.ptr(i,j) = (i + j) % 255;//grayim.ptr(i)[j]= (i + j) % 255
}
}
//创建窗口，窗口标识符为grayim
namedWindow(“grayim”, WINDOW_AUTOSIZE);
//将grayim显示在窗口上
imshow(“grayim”, grayim);
//查看第0列第i行的像素
for (int i = 0; i < 5; ++i)
{
cout << grayim.row(0).col(i) <<" " ;
}

//等待
waitKey(0);
return 0;
}
