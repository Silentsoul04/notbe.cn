淘宝商品放大镜

<!DOCTYPE html>
<html>
<head lang="en">
  <meta charset="UTF-8">
  <title>哈哈</title>
  <style>
    * {
      margin: 0;
      padding: 0;
    }

    .box {
      width: 350px;
      height: 350px;
      margin: 100px;
      border: 1px solid #ccc;
      position: relative;
    }

    .big {
      width: 400px;
      height: 400px;
      position: absolute;
      top: 0;
      left: 360px;
      border: 1px solid #ccc;
      overflow: hidden;
      display: none;
    }

    .mask {
      width: 175px;
      height: 175px;
      background: rgba(255, 255, 0, 0.4);
      position: absolute;
      top: 0px;
      left: 0px;
      cursor: move;
      display: none;
    }

    .small {
      position: relative;
    }
  </style>
</head>
<body>
<div class="box" id="box">
  <div class="small"><!--小层-->
    <img src="images/small.png" width="350" alt=""/>
    <div class="mask"></div><!--遮挡层-->
  </div><!--小图-->
  <div class="big"><!--大层-->
    <img src="images/big.jpg" width="800" alt=""/><!--大图-->
  </div><!--大图-->
</div>
<script>

  // 获取标签ID
  function my$(id) {
    return document.getElementById(id);
  }

  //获取需要的元素
  var box = my$("box");
  //获取小图的div
  var small = box.children[0];
  //遮挡层
  var mask = small.children[1];
  //获取大图的div
  var big = box.children[1];
  //获取大图
  var bigImg = big.children[0];

  // 鼠标进入显示遮挡层和大图的div
  box.onmouseover = function() {
    // 显示遮挡层
    mask.style.display = "block";
    // 显示大图的div
    big.style.display = "block";
  };

  // 鼠标离开隐藏遮挡层和大图的div
  box.onmouseout = function() {
    // 显示遮挡层
    mask.style.display = "none";
    // 显示大图的div
    big.style.display = "none";
  };

  // 鼠标的移动事件----鼠标是在小层上移动--移动鼠标--遮挡层跟着动
  small.onmousemove = function(e) {
    // 获取鼠标此时的可视区域的横纵坐标
    var x = e.clientX - mask.offsetWidth / 2;
    var y = e.clientY - mask.offsetWidth / 2;
    // x和y减100是为了抵消css设置的top和left值;
    x = x - 100;
    y = y - 100;
    // 为了防止遮挡层跑出去
    // 横纵坐标的最小值left和top
    x = x < 0 ? 0 : x;
    y = y < 0 ? 0 : y;
    // 右和下最大值  
            //如果小图的宽减遮挡层的宽小于x值就把值赋给x；
    x = x > small.offsetWidth - mask.offsetWidth? small.offsetWidth - mask.offsetWidth : x;
             //如果小图的高减遮挡层的高小于x值就把值赋给x；
    y = y > small.offsetHeight - mask.offsetHeight? small.offsetheiHht - mask.offsetHeight : y;
    // 为遮挡层赋值
    mask.style.left = x + "px";
    mask.style.top = y + "px";


    // 遮挡层的移动距离/大图的移动距离 = 遮挡层的最大移动距离/大图的最大移动距离
    // 大图的移动距离 = 遮挡层的移动距离*大图的最大移动距离/遮挡层的最大移动距离
    // 大图的横向最大移动距离
        var maxX = bigImg.offsetWidth-big.offsetWidth;
    // 大图的纵向最大移动距离
        var maxY = bigImg.offsetHeight-big.offsetHeight;

        // 大图的横向移动坐标
        var bigImgMoveX = x * maxX / (small.offsetWidth - mask.offsetWidth);
        // 大图的纵向移动坐标
        var bigImgMoveY = y * maxY / (small.offsetHeight - mask.offsetHeight);

        // 设置大图片移动
        bigImg.style.marginLeft = -bigImgMoveX + "px";
        bigImg.style.marginTop = -bigImgMoveY + "px";
    };
</script>
</body>
</html>
