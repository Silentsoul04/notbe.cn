//堆排序
viod HeadAdjust(RecordList L,int s,int m)
{
//已知L.r[s…m]中的记录除了s外，其它的均满足堆排序，使用此算法调整
t=L.r[s];
for(i=2s;i<=m;i=2)
{
if(i<m&&L.r[i]>L.r[i+1]) i++;
if(L.r[i]>L.r[s]) break;
L.r[s]=L.r[i];
s=i;
}
L.r[s]=t; //插入操作
}
//建立初始堆的过程
void CreateHeap(RecordList L)
{
for(i=L.length/2;i>=1;i–)
{
HeadAdjust(RecordList L,i,L.length);
}
}
//堆排序
void HeapSort(RecordList L)
{
CreatHeap(L);
for(i=L.length;i>=2;i–)
{
L.r[0]=L.r[1];
L.r[1]=L.r[i];
L.r[i]=L.r[0];
HeadAdjust(L,1,i-1);
}
}
