java随机生成姓名

May 12, 2019



package com.javabase.p1;

public class RandomofNames {

	/**
	 * 随机生成姓名 写的粗糙
	 * 
	 */
	public static void main(String[] args) {
		String sXing =    "赵钱孙李周吴郑王冯陈褚卫蒋沈韩杨朱秦尤许"
						+ "何吕施张孔曹严华金魏陶姜戚谢邹喻柏水窦章"
						+ "云苏潘葛奚范彭郎鲁韦昌马苗凤花方俞任袁柳"
						+"酆鲍史唐费廉岑薛雷贺倪汤滕殷罗毕郝邬安常" 
						+ "乐于时傅皮卞齐康伍余元卜顾孟平黄和穆萧尹"
						+ "姚邵湛汪祁毛禹狄米贝明臧计伏成戴谈宋茅庞"
						+ "熊纪舒屈项祝董粱杜阮蓝闵席季麻强贾路娄危"
						+ "江童颜郭梅盛林刁钟徐邱骆高夏蔡田樊胡凌霍"
						+ "虞万支柯昝管卢莫经房裘缪干解应宗丁宣贲邓"
						+ "郁单杭洪包诸左石崔吉钮龚程嵇邢滑裴陆荣翁 ";
		
		String[] sMing = {"小","明","丽","红","强","小丽","炎","燕","晓燕","洋","阳","庄","小庄","晓阳","小洋","山","进","明"
							,"小文","文","文丽","晓歌","小歌","歌","晓波","小波","小亮","晓亮","亮","品华","霞","红霞","晓炎","晓强","德华","华","城","诚","小明"
							};
					
		char[] c = null;
		c = sXing.toCharArray();

		
		for (int i = 0; i < c.length; i++) {
		//打印姓 数组
			System.out.printf("%s ",c[i]);
		}
		System.out.println("\n");
		for (int i = 0; i < sMing.length; i++) {
			//打印名 数组
			System.out.printf("%s ",sMing[i]);
		}
		System.out.println("\n");
		
		//姓 数组 长度
		int ixing = c.length;
		//名 数组 长度
		int iming = sMing.length;
		double d = 0;
		
	
		//字符串数组
		// StringBuffer[] sxmbuf = new StringBuffer[ixing];
		 StringBuffer[] sxmbuf = new StringBuffer[10000];
	
		 //为字符串数组分配空间
		 for (int i = 0; i < sxmbuf.length; i++) {
			 StringBuffer kk = new StringBuffer();
			 sxmbuf[i] = kk;
		}
		 //计算消耗cpu时间
		 long startTime= System.currentTimeMillis();
		 //循环生成随机数组装字符串
		 for (int i = 0; i < sxmbuf.length; i++) {
			 d = Math.random();
				//姓 数组中的随机数
				 int r = (int)(d * 100000) % ixing;
				 //名 数组中的随机数
				 int r1 = (int)(d * 100000) % iming;
				 
				 //组装姓名
			 String s = String.valueOf(c[r]) + sMing[r1]; 
			
			 //加到数组中StringBuffer
			 sxmbuf[i].append(s);
			 //打印组装好的 名字
			 System.out.printf("%s %d %d -------- %s","随机数: ",r,r1,sxmbuf[i].toString());
			 System.out.println("\n");
				 
		}
		 long endTime= System.currentTimeMillis();
		 System.out.println("\n");
		 //计算消耗cpu时间
		 System.out.printf("耗时 %d 毫秒",endTime - startTime);
	 
	

	}

}
