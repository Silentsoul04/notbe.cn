在虚拟机中创建kvm

May 12, 2019

一、需求
1、一台虚拟机------controller
2、centos镜像文件
二、创建步骤
#mkdir /opt/centos //创建centos文件夹
#mount /dev/sr0 /opt/centos //挂载镜像到centos文件夹
#rm -rf /etc/yum.repo/* //清空yum文件夹
#vi /etc/yum.repo/local.repo //创建yum源
[centos]
name=centos
baserul=file:///opt/centos
gpgcheck=0
#yum clean all //清空yum 缓存
#yum repolist //查看yum
#sed -i ‘s/ONBOOT=on/ONBOOT=yes/’ /etc/sysconfig/network-scripts/ifcfg-eth1 //交互式修改eth1网卡文件
#ifdown eth1 && ifup eth1 //重启eth1网卡
#yum install -y qemu-kvm libvirt virt-install bridge-utils 安装kvm包
#sed -i ‘s/^#vnc_listen/vnc_listen/’ /etc/libvirt/qemu.conf //交互式修改文件
#systemctl restart network //重启网咯
#virsh iface-bridge eth1 br0 --no-stp no–start //将eth1网卡添加到网桥
#systemctl restart network //重启网咯
#brctl show //查看网桥
#virsh-install -n vm1 -r 64 -w bridge=br0 --disk /opt/cirros-0.3.0-x86-disk.img --import --noreboot 创建kvm虚拟机
#virsh list --all //查看kvm虚拟机状态
#virsh start vm1 //启动vm1虚拟机
#virsh console vm1 //切换vm1虚拟机
#virsh-install -n vm2 -r 64 -w bridge=br0 --disk /opt/cirros-0.3.0-x86-disk-2.img --import --noreboot //创建第二台kvm虚拟机
