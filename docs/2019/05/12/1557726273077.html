北京师范大学第十七届程序设计竞赛决赛 D选数字(数学)

May 12, 2019

分析：给定一组数，求最小数的数学期望和一个组合数的乘积。
通过分析，可以求得公式： a i ∗ C ( k − 1 , n − i ) , i = 1 , 2 , 3..... n − k + 1 ai*C(k-1,n-i),i=1,2,3.....n-k+1 ai∗C(k−1,n−i),i=1,2,3.....n−k+1

怎么就wa了呢。

#include<iostream>
#include<algorithm>
#include<cstring>
#define mmod 1000000007
using namespace std;
typedef long long ll;
int T,n,k;
int a[100000];
ll tran(int m,int n){
    ll ans=1;
    for(int i=1;i<=m;i++){
        ans=ans*(n-i+1)/i;
    }
    return ans;
}
int main(){
    std::ios::sync_with_stdio(false);
    cin>>T;
    while(T--){
        ll tmp=0;
        memset(a,0,sizeof(a));
        cin>>n>>k;
        for(int i=1;i<=n;i++){
            cin>>a[i];
        }
        sort(a+1,a+n+1);
        for(int i=1;i<=n-k+1;i++){
            tmp=tmp+a[i]*tran(k-1,n-i);
        }
        cout<<tmp%mmod<<endl;
    }
    
    return 0;
}
