二维数组的鞍点

May 12, 2019

#include<stdio.h>
int main()
{
int i, j, min, m, n, p, q, max;
m = 0; n = 0; p = 0; q = 0;
int a[4][4];
for (i = 0; i < 4; i++) //给二维数组赋值并且输出二维数组
{
for (j = 0; j < 4; j++)
{
scanf_s("%d", &a[i][j]);
printf("%d\t", a[i][j]);
}
printf("\n");
}
printf("\n");
max = a[0][0]; //定义最大值为数组第一行第一列
min = a[0][0]; //定义最小值为数组第一行第一列
for (i = 0; i<4; i++)
{
m = i;
n = 0;
p = 0;
q = 0;
max = a[i][0];
for (j = 0; j<4; j++) //求出每行的最大值
if (a[i][j]>max)
{
max = a[i][j];
n = j;
}
min = a[0][n];
q = n;
for (j = 0; j<4; j++) //求出没列的最小值
if (a[j][n]<min)
{
min = a[j][n];
p = j;
}
if ((p == m)) //在行中是最大值列中最小值同时满足就是鞍点
{
printf(“鞍点为：%d\n”, a[m][n]);
printf(“此时m=%d; n=%d\n”, m, n);
break;
}
else //否则没有鞍点
if (i >= 3)
printf(“no an dian”);
}
getchar();
getchar();
return 0;
}
