3、乾坤大挪移

<!DOCTYPE html>
<html>
	<head>
		<meta charset="UTF-8">
		<title>后盾人 人人做后盾</title>
		
		<style>
			#hd1{
			width:400px;
			height:400px;
			border:4px solid orange;
		}
			#hd2{
				width:400px;
				height:400px;
				border:4px solid lightblue;
			}
		</style>
		
		<script type="text/javascript">
			window.onload=function(){
//				抓元素
				var hd1=document.getElementById("hd1");
				var hd2=document.getElementById("hd2");
				var btn=document.getElementById("btn");
				
				btn.onclick=function(){
//					获得hd1的内容
					var con=hd1.innerHTML;
//					将con的值写入到hd2中
					hd2.innerHTML=con;
//					将hd1清空
					hd1.innerHTML='';
				}
				
			}
		</script>
		
		
	
	</head>
	<body>
		
	<div id="hd1">后盾人 人人做后盾！后盾人 人人做后盾！后盾人 人人做后盾！
	<img src="dan.jpg" alt="" />

	<br /><br />
	<div id='hd2'></div>	
	</body>
</html>
