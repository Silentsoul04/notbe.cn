URL编程

为了表示URL，java.net 中实现了类 URL。我们可以通过下面的构造器来初始化一个 URL 对象：
public URL (String spec)：通过一个表示URL地址的字符串可以构造一个URL对象。例如：URL url = new URL (“http://www. baidu.com/”);

package com.ghl.demo;

import java.io.IOException;
import java.io.InputStream;
import java.io.OutputStream;
import java.net.MalformedURLException;
import java.net.URL;
import java.net.URLConnection;

import org.junit.Test;

/**
 *URL统一资源定位符，一个URL的对象对应这互联网的一个资源，该URL是一个类
 *可以通过URL的对象调用其相应的方法来将此资源读取或下载
 */
public class TestURL {

	public static void main(String[] args) {
		
	}
	@Test
	public void testUrlMethod() throws Exception {
		URL url=new URL("http://localhost:8080/examples/myTest.txt");
		/*
		 * public String getProtocol(  )     获取该URL的协议名
			public String getHost(  )           获取该URL的主机名
			public String getPort(  )            获取该URL的端口号
			public String getPath(  )           获取该URL的文件路径
			public String getFile(  )             获取该URL的文件名
			public String getRef(  )             获取该URL在文件中的相对位置
			public String getQuery(   )        获取该URL的查询名
		 */
		System.out.println(url.getProtocol());
		System.out.println(url.getHost());
		System.out.println(url.getPort());
		System.out.println(url.getPath());
		System.out.println(url.getFile());
		System.out.println(url.getRef());
		System.out.println(url.getQuery());
		
		/*
		 * 如何将服务端的资源读取进来
		 */
		InputStream is = url.openStream();//通过openStream()方法拿到输入流对象
		byte[] b=new byte[1024];
		int len;
		while((len=is.read(b))!=-1) {
			String str=new String(b,0,len);
			System.out.println(str);
			
		}
		is.close();
	}
	/**
	 * 既有数据的输入又有数据的输出可以使用URlConllelction类
	 * @throws IOException 
	 */
	@Test
	public void testUrlConllection() throws IOException {
		URL url=new URL("http://localhost:8080/examples/myTest.txt");
		URLConnection openConnection = url.openConnection();
		InputStream inputStream = openConnection.getInputStream();//输入流
		OutputStream outputStream = openConnection.getOutputStream();//输出流
		
		byte[] b=new byte[1024];
		int len;
		while((len=inputStream.read(b))!=-1) {
			outputStream.write(b,0,len);
		}
		
	}
}
