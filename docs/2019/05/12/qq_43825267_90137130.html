欧拉函数

欧拉函数是小于n的正整数中与n互质的数的数目（φ(1)=1）。（例如φ(8)=4，因为1,3,5,7均和8互质。）

其中p1, p2……pn为x的所有质因数，x是不为0的整数。
质因数：质因数（素因数或质因子）在数论里是指能整除给定正整数的质数。

int ouler(int n)
{
    int number = 1, i;
    for(i=2; i*i <= n; i++)
    {
        if(n%i == 0)
        {
            n =n/ i;
            number *= (i-1);
            while(n%i == 0)
            {
                n /= i;
                number *= i;
            }
        }
    }
    if(n > 1)
        number *= (n-1);
    return number;
}

int quickpow(long long int a,long long int b,long long int p)
{
    long long int r=1,base=a;
    while(b)
    {
        if(b%2)
            r=r*base%p;
        base=base*base%p;
        b=b/2;
    }
    return r;
}

{
