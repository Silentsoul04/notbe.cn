3.dom抓元素的几种方式

<!DOCTYPE html>
<html>
	<head>
		<meta charset="UTF-8">
		<title>后盾人 人人做后盾</title>
		
		<script type="text/javascript">
			window.onload=function(){
//				1.通过id抓取元素，js抓的时候只抓第一个
				document.getElementById("one").style.color='red';
//				2.通过class抓元素
				
				
//				注意:通过id抓过来的元素只有一个,
//				但是通过class/标签名/name属性抓到的元素会以数组的形式返回回来.哪怕只抓到一个元素
				var ps = document.getElementsByClassName("two");
				ps[0].style.color='red';
				ps[1].style.color='red';
				ps[2].style.color='darkgoldenrod';
				
				
//				通过name属性抓元素
				var pss= document.getElementsByName("hd");
				pss[0].style.color='blue';
				
//				通过标签名抓元素
				var psss=document.getElementsByTagName("p");
				for(var i=0;i<psss.length;i++)
				{
					psss[i].style.color='yellow';
				}
				
		//		document.getElementById("box").getElementsByClassName('hd');   注意不能抓id的id元素
				
			}
		</script>
		
		
	
	</head>
	<body>
		
		<p>后盾人 人人做后盾！0</p>
		<p id="one">后盾人 人人做后盾！1</p>
		<p >后盾人 人人做后盾！2</p>
		<p class="two">后盾人 人人做后盾！3</p>
		<p name ='hd'>后盾人 人人做后盾！4</p>
		<p class="two">后盾人 人人做后盾！5</p>
		<p class="two">后盾人 人人做后盾！6</p>
		<p name='hd'>后盾人 人人做后盾！7</p>
		
	</body>
</html>
