1. JDBC的连接（Mysql数据库）

JDBC（java database connectivity）：java连接数据库的技术

  1. 了解市面上常用关系型和非关系型数据库

    • 关系型数据库：MySql数据库 SqlServer数据库 Oracle数据库 DB2数据库IBM
    • 非关系型数据库：Redis(缓存数据库)

  2. JDBC简单说 就是Java为我们提供操作数据库的一套完整API(类和接口) 可以对任何数据库实现无缝衔接

  3. Java语言通过数组或者说集合框架实现数据的存储和操作 数据库的数据是通过数据库客户端工具进行单独操作

  4. JDBC如何实现数据库的连接并且完成数据库数据的操作(CRUD 增删改查)的步骤：

    • （加载驱动）加载数据库的驱动（作用：通过加载数据库驱动 确定我们当前JDBC连接的是什么类型的数据库 因为不同的驱动对应不同数据库）
    • （建立连接） 建立和数据库的连接(作用：通过驱动管理器调用得到数据库连接的方法 传递数据库相关的连接参数 实现连接)
    • （创建通道 ）创建一个通道对象(作用：通过通道对象实现对数据库中的数据进行操作)
    • （处理结果） 处理结果集(作用：把数据库得到的数据进行相应处理<在程序中进行展现输出>)
    • （关闭资源） 关闭JDBC相关资源 释放内存空间
    try{
			
// 1.加载MySql数据库驱动
// 加载驱动的核心实现思路<通过反射机制>：根据传递mysql数据库完整驱动类路径 实现加载
// 异常：java.lang.ClassNotFoundException: com.mysql.jdbc.Driver
// 原因：1.完整的类路径写错了
// 2.我们从哪里去找这个类路径呢？一定要把当前驱动类路径所在的驱动包加载到当前项目中
// 注意：当加载驱动时 一定要确保数据库的驱动包是否导入到项目中
			
			Class.forName("com.mysql.jdbc.Driver");		// 从jdk5开始不需要自己手动创建（自动加载），但是为了兼容老程序，还是加上为好。
			
// 2.建立mysql数据库的连接，得到一个连接对象
			Connection conn = DriverManager.getConnection(
					"jdbc:mysql://localhost:3306/girls?useUnicode=true&characterEncoding=utf-8&useSSL=false","root","userroot");
			
// 3.通过连接对象conn去创建通道对象
			Statement stmt = conn.createStatement();
			String sql = "select * from admin";
// 根据通道对象的执行查询方法 查询指定的sql语句 返回封装了所有部门信息的（ResultSet）结果集对象
			
// 4.处理结果集对象
			ResultSet rs = stmt.executeQuery(sql);
// 遍历结果集对象 取出每一条部门记录
			System.out.println("信息是：");
			while(rs.next()){
				int id = rs.getInt("id");
				String name = rs.getString("username");
				int pwd = rs.getInt("password");
				System.out.println(id+"--"+name+"--"+pwd);
			}
			
// 5.关闭资源
			rs.close();
			stmt.close();
			conn.close();
		}catch(Exception e){
			
			System.out.println("驱动路径找不到！");
			e.printStackTrace();

  • 下面是实现了jdbc的封装的一个工具类：
package com.jdbc.www.until;

/* * 实现了jdbc的封装的一个工具类 * 1.当前类封装了jdbc API常用接口对象属性(Connection对象、PreparedStatement对象、ResultSet对象) * 2.封装获取连接的方法 * 3.封装通用的查询方法 * 4.封装通用的增删改方法 * 5.封装关闭资源的方法 * * */

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.ResultSet;

import static java.sql.DriverManager.getConnection;

public class BaseDao {

// 1.当前类封装了jdbc API常用接口对象属性(Connection对象、PreparedStatement对象、ResultSet对象)
    public Connection conn;
    public PreparedStatement pstmt;
    public ResultSet rs;

// 2.封装获取连接的方法
    public Connection getConn() throws Exception {
// Class.forName("cpm.mysql.jdbc.Driver");
        conn = DriverManager.getConnection("jdbc:mysql://localhost:3306/myemployees", "root", "userroot");
        if (conn != null || !conn.equals("")){
            return conn;
        }
        return null;
    }

// 3.定义通用的查询方法，传递两个参数，一个是String类型的sql语句，一个是保存sql的？参数的Object类型的数组
    public ResultSet getQuery(String sql,Object[] objs) throws Exception{
// 调用连接方法，连接对象
        conn = this.getConn();
// 预加载通道，预编译
        pstmt = conn.prepareStatement(sql);
// 循环遍历参数数组，完成给sql中的？的赋值操作
        for (int i = 0;i<objs.length;i++){
            pstmt.setObject(i+1,objs[i]);
        }
        rs = pstmt.executeQuery();
        return rs;
    }

    // 5.封装关闭资源的方法
    public void close(){
        try{
            if (rs != null){
                rs.close();
            }
            if (pstmt != null){
                pstmt.close();
            }
            if (conn != null){
                conn.close();
            }

        }catch(Exception e){
            e.printStackTrace();
        }
    }
}
  • 下面是定义dept中简单查询的一些方法
package com.jdbc.www.dao;

import com.jdbc.www.jdbc.Dept;
import com.jdbc.www.until.BaseDao;

import java.util.ArrayList;
import java.util.List;

public class DeptDao extends BaseDao {

// 定义查询所有部门的方法
// 封装到集合中
    public List<Dept> getDepts() throws Exception{
// 一个sql语句
        String sql = "select * from departments";
// 一个Object数组
        Object[] objs = {};
// 调用BaseDao的getQuery方法，结果集来承载
        rs = this.getQuery(sql,objs);
// 定义一个集合，存储查询的结果集
        List<Dept> depts = new ArrayList<>();
// 遍历结果集
        while (rs.next()){
           Dept dept =  new Dept(rs.getInt(1),rs.getString(2),rs.getInt(3),rs.getInt(4));
           depts.add(dept);
        }
// 关闭资源
        this.close();
// 返回
        return depts;
    }

// 定义根据部门名称进行模糊查询部门信息
    public List<Dept> getLikeDepts() throws Exception{
        String sql = "select * from departments where department_name like ?";
        Object[] objs = {"%M%"};
// 调用BaseDao的getQuery方法，结果集来承载
        rs = this.getQuery(sql,objs);
// 定义一个集合，存储查询的结果集
        List<Dept> depts = new ArrayList<>();
// 遍历结果集
        while (rs.next()){
            Dept dept =  new Dept(rs.getInt(1),rs.getString(2),rs.getInt(3),rs.getInt(4));
            depts.add(dept);
        }
// 关闭资源
        this.close();
// 返回
        return depts;

    }

}
