    static void push(int n) {
        useStack.push(n);
        if (withStack.empty()) {
            withStack.push(n);
            return;
        }
        if (n <= withStack.peek()) {
            withStack.push(n);
        } else {
            withStack.push(withStack.peek());
        }

    }

    static int myPop() {
        if (!withStack.empty() && !useStack.empty()) {
            return withStack.pop();
        }
        return 0;
