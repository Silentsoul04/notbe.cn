数据结构 —— 队列

May 12, 2019

什么是队列

  • 队列是一种操作受限的线性表，队列只允许在表的一端进行插入，在表的另一端进行删除。可进行插入的一段称为队尾，可进行删除的一端称为队头。
  • 队列的主要特点就是先进先出。依照存储结构可分为：顺序队和链式队。

顺序队列


其中，循环队列必须损失一个存储空间（如上图中下标为0的空间），如果控件0也存入元素，则队满和队空的条件都是q.front==q.rear，此时便无法区分队空和队满了。

  1. 队空状态：q.front==q.rear
  2. 队满状态：(q.rear+1)%maxSize==q.front

顺序队列定义

// 顺序队列定义
typedef struct 
{
	int data[maxSize];
	int front;	// 队首指针
	int rear;	// 队尾指针
}SqQueue;

初始化队列

// 初始化队列
void initQueue(SqQueue &q)
{
	q.front=q.rear=0;
}

判断出队

// 判断队空
bool isQueueEmpty(SqQueue q)
{
	if(q.front==q.rear)
		return true;
	else
		return false;
}

判断队满

// 判断队满
bool isQueueFull(SqQueue q)
{
	if((q.rear+1)%maxSize==q.front)
		return true;
	else
		return false;
}

进队

// 进队
int enQueue(SqQueue &q,int x)
{
	if(isQueueFull(q))
		return 0;	// 如果队满，则无法进队
	q.rear=(q.rear+1)%maxSize;
	q.data[q.rear]=x;	// 先移动指针再存入元素
	return 1;
}

出队

// 出队
int deQueue(SqQueue &q,int &x)
{
	if(isQueueEmpty(q))
		return 0;	// 如果队空，则无法出队
	q.front=(q.front+1)%maxSize;
	x=q.data[q.front];
	return 1;
}

链队

  1. 栈空状态：q->front==q->rear
  2. 栈满状态：假设内存无限大，此时不存在栈满的情况

队结点类型定义

// 队结点类型定义
typedef	struct LNode
{
	int data;
	struct LNode *next;
}LNode;

链队类型定义

// 链队类型定义
typedef struct 
{
	LNode *front;	// 队头指针
	LNode *rear;	// 队尾指针
}LQueue;

初始化链队

// 初始化链队
void initQueue(LQueue *&q)
{
	q->front=q->rear=(LNode*)malloc(sizeof(LNode));
	q->front->next=NULL;	// q->front其实就相当于头结点
}

判断队空

// 判断队空
bool isQueueEmpty(LQueue *q)
{
	if(q->rear==q->front)
		return true;
	else
		return false;
}

入队

// 入队
void enQueue(LQueue *&q,int x)
{
	LNode *p;
	p=(LNode*)malloc(sizeof(LNode));
	p->data=x;
	p->next=NULL;
	q->rear->next=p;	// 将新结点与原来的队尾结点连接
	q->rear=p;	// 将队尾结点改为该新结点
}

出队

// 出队
int deQueue(LQueue *&q,int &x)
{
	if(isQueueEmpty(q))
		return 0;
	p=q->front->next;
	x=p->data;
	q->front->next=p->next;
	if(q->rear==p)	//如果仅剩下一个结点
		q->rear=q->front;
	free(p);
	return 1;
}
