struct函数结构数组的应用

May 12, 2019

	借鉴	C primer Plus P444,结构数组入门应用 
#include "pch.h"
#include <iostream>
#include <string.h>
char *s_gets(char*st, int n);
#define MAXTITL 40
#define MAXAUTL 40
#define MAXBXS 40
struct book
{
	char title[MAXTITL];
	char auther[MAXAUTL];
	float value;
};
int main()
{
	struct book library[MAXBXS];
	int count = 0;
	int index;
	printf("please enter the book title \n");
	printf("please [enter] at the start of a line to stop. \n");
	while (count<MAXBXS&&s_gets(library[count].title,MAXTITL)!=NULL&&library[count].title[0]!='\0')
	{
		printf("Enter the auther.\n");
		s_gets(library[count].auther, MAXAUTL);
		printf("Enter the title\n");
		scanf_s("%f", &library[count++].value);
		while (getchar()!='\n')
				continue;
		if (count<MAXBXS)
		printf("Enter the next title\n");
	}
	if (count>0)
	{
		printf("Here is the list of your books:\n");
		for (size_t index = 0; index < count; index++)
			printf("%s by %s:$%.2f\n", library[index].title, library[index].auther, library[index].value);
	}
	else
	{
		printf("No books,too bad\n");
	}
	return 0;
}
char *s_gets(char*st, int n)
{
	char *hua;

	if (hua)
	{

			i++;

		}
		else
		{
			while (getchar() != '\n')
			{

			}
		}

	}
	return hua;
}
