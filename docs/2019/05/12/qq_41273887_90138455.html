///poj-3159 Candies 模板题
/**
1.差分约束系统
2.cincout导致tle
3.memset(void *s, int ch,size_t n);中ch实际范围应该在0~~255，
因为该函数只能取ch的后八位赋值给你所输入的范围的每个字节
4.memset(a,-1,sizeof a);全为-1
5.fill(a,a+100,inf);绝对是我自己的问题
6.卡vector，内存管理机制会超时
7.数组实现邻接表Oops！
**/



#include <iostream>
#include <string.h>
#include <stdio.h>
#include <algorithm>
#include <queue>
using namespace std;
const int MAX_V=30005,MAXed=150010;
const int INF=0x3f3f3f3f;
typedef pair<int,int>P;
int V,m;
int edg;
int first[MAX_V],next[MAXed],u[MAXed],v[MAXed],w[MAXed];
int d[MAX_V];
priority_queue<P,vector<P>,greater<P> >que;
void add(int pu,int pv,int pw)
{
    u[edg]=pu;
    v[edg]=pv;
    w[edg]=pw;
    next[edg]=first[pu];
    first[pu]=edg;
    edg++;
}
void dijkstra(int s)
{
	int vis[MAX_V];
    for(int i=0;i<=V;i++)
    {
        d[i]=INF;
        vis[i]=0;
    }
	d[s]=0;
	que.push(P(0,s));
	while(!que.empty())
	{
		P p=que.top();
		que.pop();
		int x=p.second;
		if(vis[x]||d[x]<p.first)continue;
        vis[x]=1;
		for(int e=first[x];e!=-1;e=next[e])
        {
            if(d[v[e]]>d[x]+w[e])
            {
                d[v[e]]=d[x]+w[e];
                que.push(make_pair(d[v[e]],v[e]));
            }
        }
	}
}
int main()
{
    scanf("%d%d",&V,&m);
    int st,dt,vt;
    memset(first,-1,sizeof(first));
    while(m--)
    {
        scanf("%d%d%d",&st,&dt,&vt);
        add(st,dt,vt);

    }
    dijkstra(1);
    printf("%d",d[V]);
}
