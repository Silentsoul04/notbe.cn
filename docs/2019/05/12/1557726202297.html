/********
问题描述
给出n个数，找出这n个数的最大值，最小值，和。

输入格式
第一行为整数n，表示数的个数。

第二行有n个数，为给定的n个数，每个数的绝对值都小于10000。

输出格式
输出三行，每行一个整数。第一行表示这些数中的最大值，第二行表示这些数中的最小值，第三行表示这些数的和。
样例输入
5
1 3 -2 4 5
样例输出
5
-2
11
数据规模与约定
1 <= n <= 10000

********/

#include<iostream>
using namespace std;

int  main(void)
{
    int max,min ,n,he=0;
    int num[100];
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>num[i];
    }

    max=num[0];
    min=num[0];
    for(int i=0;i<n;i++)
    {
        he+=num[i];  //和
        if(max<num[i]){max=num[i];}  //最大值
        if(min>num[i]){min=num[i];}   //最小值
    }

    cout<<max<<endl;
    cout<<min<<endl;
    cout<<he<<endl;
}
