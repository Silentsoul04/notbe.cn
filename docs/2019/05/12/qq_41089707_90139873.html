解1 字典序

注意 切片 浅拷贝

class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        
        n = len(nums)
        
        count = 1
        for i in range(1, n + 1):
            count *= i
        
        result = list()
        for k in range(count):
            
            part = n
            for i in range(n - 1, 0, -1):
                if nums[i] > nums[i - 1]:
                    part = i - 1
                    break

            if part == n:
                nums[:] = nums[::-1]
                result.append(nums[:])
                continue

            for j in range(n - 1, part, -1):
                if nums[j] > nums[part]:
                    nums[part], nums[j] = nums[j], nums[part]
                    break
                    
            nums[part + 1 : n] = nums[part + 1 : n][::-1]
			
			# 注意为 nums[:] 不能写为nums 
            result.append(nums[:])
            
        return result

class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        
        result = list()
        
        stack = list()
        
        self.dfs(nums, result, stack)
        
        return result
        
        
        
        
    def dfs(self, nums, result, stack):
        
        if len(nums) == 0:
            result.append(stack[:])
            
        else:
            
            for i in range(len(nums)):
                
                stack.append(nums[i])
                
                del nums[i]
                
                self.dfs(nums, result, stack)
                
                nums.insert(i, stack.pop())

解3

递归 深度优先

思路：
n个数 第1个位置有n个选择，第2个位置有n-1个选择…第n个位置有1个选择；
递归之前交换位置，就是在选择一个数坐在当前位置
递归之后交换位置，就是在还原最开始的顺序

例：
[1, 2, 3]
深度优先遍历如下：

  • i = 0
    • j = 0
    • 选择1坐在第一个位置 [1, 2, 3]
    • i = 1
      • j = 1
      • 选择2坐在第2个位置
      • i = 2
        • j = 2
        • 选择3坐在第3个位置
        • start + 1 == end; result.append([1, 2, 3])
        • 交换 i = 2, j = 2 位置 还原数组
      • 交换 i = 1, j = 1 位置 还原数组
    • 交换 i = 0, j = 0 位置 还原数组
    • j = 1
    • 选择2坐在第1个位置
    • i = 1
      • j = 1
      • 选择1坐在第2个位置
      • i = 2
        • j = 2
        • 选择3坐在第3个位置
        • start + 1 == end; result.append([2, 1, 3])
        • 交换 i = 2, j = 2 位置 还原数组
      • 交换 i = 1, j = 1 位置 还原数组
    • 交换 i = 0, j = 1 位置 还原数组
    • …
class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        
        result = list()
        
        n = len(nums)
        
        self.dfs(nums, 0, n, result)
        
        return result
        
        
        
    def dfs(self, nums, start, end, result):
        
        if start == end:
            result.append(nums[:])
            
        else:
            i = start
            for j in range(start, end):
                nums[i], nums[j] = nums[j], nums[i]
                self.dfs(nums, start + 1, end, result)
                nums[i], nums[j] = nums[j], nums[i]
