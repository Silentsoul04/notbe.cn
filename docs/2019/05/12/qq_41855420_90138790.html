LeetCode 旋转数字（多种算法实现）

May 12, 2019

我们称一个数 X 为好数, 如果它的每位数字逐个地被旋转 180 度后，我们仍可以得到一个有效的，且和 X 不同的数。要求每位数字都要被旋转。

如果一个数的每位数字被旋转以后仍然还是一个数字， 则这个数是有效的。0, 1, 和 8 被旋转后仍然是它们自己；2 和 5 可以互相旋转成对方；6 和 9 同理，除了这些以外其他的数字旋转以后都不再是有效的数字。

现在我们有一个正整数 N, 计算从 1 到 N 中有多少个数 X 是好数？
示例:

输入: 10
输出: 4
解释: 
在[1, 10]中有四个好数： 2, 5, 6, 9。
注意 1 和 10 不是好数, 因为他们在旋转之后不变。

注意:

N 的取值范围是 [1, 10000]。

思 路 分 析 ： \color{blue}{思路分析：} 思路分析：这道题比较简单，只要按照要求进行旋转每一位即可。

class Solution {
public:
    int rotatedDigits(int N) {
        int resCount = 0;
        vector<int> rotateNums = {0, 1, 5, -1, -1, 2, 9, -1, 8, 6};//每个数字旋转后的数字，等于-1说明这个数字无法旋转
        for (int num = 1; num <= N; ++num){
            stack<int> rotateStack;
            int tempNum = num, nextRotate;
            //先将tempNum中从低到高位逐渐旋转，放入栈中
            while (tempNum != 0){
                nextRotate = rotateNums[tempNum % 10];
                rotateStack.push(nextRotate);
                if (nextRotate == -1){//出现负一，说明这个数字无法旋转
                    break;
                }
                tempNum /= 10;
            }
            if (rotateStack.top() != -1){
                //然后从栈中把数字读出来（注意顺序）
                int rotateNum = 0;
                while (!rotateStack.empty()){
                    rotateNum = rotateNum * 10 + rotateStack.top();
                    rotateStack.pop();
                }
                if (rotateNum != num){
                    resCount += 1;
                }
            }
        }
        return resCount;
    }
};


时间复杂度、空间复杂度都很高。。。
优化思路：由于上面的算法涉及到大量的求余、进出栈，所以效率非常低下。之所以我们使用栈就是利用反转的数字去构造出整体反转后的数字，蛋式这是没有必要的。

class Solution {
public:
    int rotatedDigits(int N) {
        int resCount = 0;
        vector<int> rotateNums = {0, 1, 5, -1, -1, 2, 9, -1, 8, 6};//每个数字旋转后的数字，等于-1说明这个数字无法旋转
        for (int num = 1; num <= N; ++num){
            bool isRote = false;//用于记录是否有数字反转后不是自己
            int tempNum = num, nextRotate;
            //先将tempNum中从低到高位逐渐旋转，放入栈中
            while (tempNum != 0){
                nextRotate = rotateNums[tempNum % 10];
                if (nextRotate == -1){//出现负一，说明这个数字无法旋转
                    break;
                }
                if (nextRotate != tempNum % 10){//出现反转后不是自身的数字
                    isRote = true;
                }
                tempNum /= 10;
            }
            //只有当tempNum == 0说明这个数的所有数位都可翻转，isRote说明出现数位反转后不是本身，即整体反转后不是本身
            if (tempNum == 0 && isRote){
                resCount += 1;
            }
        }
        return resCount;
    }
};


当然也可以转换为字符串进行翻转操作。

class Solution {
public:
    int rotatedDigits(int N) {
        int resCount = 0;
        vector<char> rotateChs = {'0', '1', '5', 'a', 'a', '2', '9', 'a', '8', '6'};//每个数字旋转后的数字，等于'a'说明这个数字无法旋转
        for (int num = 1; num <= N; ++num){
            char rotateCh;
            string tempNum = to_string(num);//转换为字符串
            bool isRote = false;//用于记录是否有数字反转后不是自己
            for (auto ch : tempNum){//对每个字符进行旋转
                rotateCh = rotateChs[ch - '0'];
                if (rotateCh == 'a'){//出现无法旋转的字符
                    break;
                }
                if (ch != rotateCh){//出现反转后不是自身的数字
                    isRote = true;
                }
            }
            //只有当rotateCh != 'a'说明这个数的所有数位都可翻转，isRote说明出现数位反转后不是本身，即整体反转后不是本身
            if (rotateCh != 'a' && isRote){
                resCount += 1;
            }
        }
        return resCount;
    }
};
