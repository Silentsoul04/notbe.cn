Leetcode 2.两数相加
题目
给出两个非空的链表用来表示两个非负的整数。其中，它们各自的位数是按照逆序的方式存储的，并且它们的每个节点只能存储一位数字。如果，我们将这两个数相加起来，则会返回一个新的链表来表示它们的和。您可以假设除了数字 0 之外，这两个数都不会以 0 开头。
下面是leetcode给出的ListNode结构

public class ListNode {
	 public int val;
	 public ListNode next;
	 public ListNode(int x) { val = x; } //构造函数

}

示例：
输入：(2 -> 4 -> 3) + (5 -> 6 -> 4)
输出：7 -> 0 -> 8
原因：342 + 465 = 807

题解
按照逆序的方法存储，两数相加的时候正好可以从ListNode的第一个结点开始向后相加（即第一个结点为个位，第二个为十位）
代码如下

/* * public class ListNode { public int val; public ListNode next; public ListNode(int x) { val = x; } //构造函数 } */

class Solution {
	public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
    	ListNode result = new ListNode(0);
    	ListNode a,b;
    	a=result;
    	ListNode p=l1, q=l2;
    	int carry=0;
    	while(p!=null||q!=null||carry!=0)
    	{
    		int x,y;
    		if(p!=null)
    		{
    			x=p.val;
    		}
    		else {
    			x=0;
    		}
    		if(q!=null)
    		{
    			y=q.val;
    		}
    		else {
    			y=0;
    		}
    		int r =(x+y+carry)%10;           //结果序列当前位置数值
    		if((x+y+carry)==((x+y+carry)%10))
    		{
    			carry =0;
    		}
    		else {
    			carry=1;
    		}
    		if (p != null) p = p.next;
            if (q != null) q = q.next;
            
            b = new ListNode(r);
            a.next=b;
            a=b;
        	}
		
        return result.next;
    }
   /* public static void main(String[] args) { ListNode l1 =new ListNode(2); l1.next=new ListNode(4); l1.next.next = new ListNode(3); ListNode l2 =new ListNode(5); l2.next=new ListNode(6); l2.next.next = new ListNode(4); Solution s =new Solution(); ListNode result=s.addTwoNumbers(l1, l2); System.out.println(result.next.next.next.val); }*/
}
