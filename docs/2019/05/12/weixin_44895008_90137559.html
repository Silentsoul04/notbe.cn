package com;
//外部类只能用 public 或者 默认的
public class User {
private String userName;
public void showinfo(){
System.out.println(“userName:”);
}
//内部类都可以(修饰符)
// 即使是内部类，一样可以写属性和方法
// 内部类和外部类的 属性和方法一个级别，要先new外部类才有内部类
public class inner_class{
public String innerName;
public void info(){
System.out.println(innerName);
}
}
//静态内部类 不用创建User对象 可以直接 new 内部静态类
public static class inner_static_class{
public void showinfo(){
System.out.println(“inner_static_class”);
}
}

public void inner_class_method(){

//局部内部类，只能在当前方法中有效
class inner_static_class{
public void showinfo(){
System.out.println(" class inner_static_class{\n");
}
}
inner_static_class inner_static_class = new inner_static_class();
inner_static_class.showinfo();

}
