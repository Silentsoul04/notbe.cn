/*******************IIC Start******************/
              void IIC_Start()
              {
               SCL=high;       //Delay_us(1);        
               SDA=high;
               _nop_();
               SDA=low;
               _nop_();_nop_();
               SCL=low;
              }
/*************IIC Stop*************************/
              void IIC_Stop()
              {
              SDA=low;
              _nop_();
              SCL=high;
              _nop_();_nop_();
              SDA=high;
              }
/*****************IIC Write byte***************/
              bit Write_IIC_Byte(unsigned char IIC_Byte)
              {
                             unsigned char i;
                             bit Ack_Bit;                    //应答信号
                                for(i=0;i<8;i++)        
                                {
                                if(IIC_Byte & 0x80)        //1?0?
                                SDA=high;
                                else
                                SDA=low;   //Delay_us(1);
                              SCL=high;
                              _nop_();_nop_();
                              SCL=low;   //Delay_us(1);
                              IIC_Byte<<=1;            //loop
                                }
                                SDA=high;                       //释放IIC SDA总线为主器件接收从器件产生应答信号    
                                _nop_();_nop_();
                                SCL=high;                     //第9个时钟周期
                              _nop_();_nop_();
                                 Ack_Bit=SDA;                   //读取应答信号
                                SCL=low;
                              return Ack_Bit;    
              }
