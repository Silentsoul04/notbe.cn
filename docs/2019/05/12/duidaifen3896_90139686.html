using namespace std;

template <class T>
int count(T a[], int n, const T& value)
{
    int theCount = 0;
    for (int i = 0; i < n; i++)
        if (a[i] == value)
            theCount++;
    return theCount;
}

{
    int x[5]={3,2,3,4,5};

题10

#include <iostream>
int abc(int a,int b,int c);
using namespace std;
int main()
{
    try {
        //cout << abc(2,1,2)<<endl;
        cout << abc(0,0,0)<<endl;
        cout << abc(-2,-2,-2)<<endl;

    }
    catch(exception& e)//捕获基类型exception以及其派生的类型（例如bad_alloc和ba_typeid）
    {
        cout <<"a"<<endl;
    }

    catch(int e)
    {
        if(e==2)
            cout<<"this is 2"<<endl;
        if(e==1)
            cout<<"e==1"<<endl;
    }
    catch(...)//捕捉异常，不管什么类型
    {
        cout<<"..."<<endl;
    }
    return 0;
}

int abc(int a,int b,int c)
{
    if(a<0&&b<0&&c<0)
        throw 1;
    if(a==0&&b==0&&c==0)
        throw 2;
    return a+b*c;
}

结果：

发生异常，程序终止，所以只会输出第一个异常。
