子程序-输入成绩

May 12, 2019

源码:

setcore proc
    push bx
    push cx 
    push dx
                
    mov dx, 000ah
    xor bl, bl
    mov cx, 0003h
lo:    
    call STDIN   ;输入子程序 
    cmp al,0dh
    jz  ot1
    call ASCII2B  ;ASCII转数值子程序
    ;实现bl* 10 + al
    push ax    ;将al压入栈
    mov al, bl ;将bl中的值放入al
    mul dl    ;  bl * 10
    pop bx     ;al 的值放入 bl中
    add al, bl ;bl * 10 + al
    mov bl, al ;将al放入bl中
    loop lo
ot1:
    mov al, bl  ;将bl的值放入到al中
    pop dx
    pop cx
    pop bx   
    ret
setcore endp  

作用：从屏幕上读入分数（1-3位）。

出口参数：al（8位二进制储存）
