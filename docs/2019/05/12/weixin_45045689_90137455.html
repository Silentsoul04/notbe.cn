Train Problem I hdu 1022

May 12, 2019

题目链接

题意
用输入一种状态，用栈是否可以转为另一种排法

题解：
如果可以，要输出操作。我直接又模拟一遍，输出操作。导致用的时间较多。
可以标记在第一遍模拟时的操作的！如入栈标记为1，出栈标记为0.

类似题：
链接

代码：

#include<iostream>
#include<string>
#include<stack>
using namespace std;
int main()
{
	int N;
	while (cin >> N)
	{
		string o1, o2;
		stack<int> s;
		int t(0);
		cin >> o1 >> o2;
		for (int i = 0; i < N; i++)
		{
			s.push(o1[i] - '0');
			while (s.empty() != 1 && s.top() == o2[t] - '0')
			{
				s.pop();
				t++;
			}
		}
		if (t == N)
		{
			cout << "Yes." << endl;
			t = 0;
			for (int i = 0; i < N; i++)
			{
				s.push(o1[i] - '0');
				cout << "in" << endl;
				while (s.empty() != 1 && s.top() == o2[t] - '0')
				{
					s.pop();
					cout << "out" << endl;
					t++;
				}
			}
			cout << "FINISH" << endl;
		}
		else
		{
			cout << "No." << endl;
			cout << "FINISH" << endl;
		}
	}
	return 0;
}
