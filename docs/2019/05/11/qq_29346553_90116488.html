四、已知特征值，如何求相似度

int CalcSimilarity(float fc1[256], float fc2[256])
{
int dim = 256;

return trunc((simd_dot(fc1, fc2, dim) / (sqrt(simd_dot(fc1, fc1, dim))* sqrt(simd_dot(fc2, fc2, dim)))) * 10000);

}

float simd_dot(const float* x, const float* y, const long& len) {
float inner_prod = 0.0f;
__m128 X, Y; // 128-bit values
__m128 acc = _mm_setzero_ps(); // set to (0, 0, 0, 0)
float temp[4];

long i;
for (i = 0; i + 4 < len; i += 4) {
	X = _mm_loadu_ps(x + i); // load chunk of 4 floats
	Y = _mm_loadu_ps(y + i);
	acc = _mm_add_ps(acc, _mm_mul_ps(X, Y));
}
_mm_storeu_ps(&temp[0], acc); // store acc into an array
inner_prod = temp[0] + temp[1] + temp[2] + temp[3];

// add the remaining values
for (; i < len; ++i) {
	inner_prod += x[i] * y[i];
}
return inner_prod;

}
