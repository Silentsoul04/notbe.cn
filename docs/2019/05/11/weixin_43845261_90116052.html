5362: C++实验：STL之vector

描述：

将n个数字输入到vector里，随后输出。
部分代码已经给出，请补充完整，提交时请勿包含已经给出的代码。

 int main()
{
    vector<int> vec;
    Input(vec);
    for(int i=0;i<vec.size();i++)
    {
        cout<<vec[i]<<endl;
    }
    return 0;
}

输入：

第一行为正整数n。
第二行为n个整数，空格隔开。

输出：

输出输入的n个整数，每行一个。

样例：

3
1 2 3

1
2
3

#include<iostream>
#include<vector>
using namespace std;
void Input(vector<int> &qy)
{
	int n,i,a;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a;
		qy.push_back(a);
	}
}

5363: C++实验：STL之vector2

描述：

依次输入n个整数，每次输入时检查该值是否已经出现在vector中，如果存在则不插入，否则将其插入到开头位置。
部分代码已经给出，请补充完整，提交时请勿包含已经给出的代码。

 int main()
{
    vector<int> vec;
    int n, x;
    cin>>n;
    while(n--)
    {
        cin>>x;
        Insert(vec, x);
    }
    for(vector<int>::iterator it=vec.begin();it!=vec.end();it++)
    {
        cout<<*it<<endl;
    }
    return 0;
}

输入：

第一行为正整数n。
第二行为n个整数，空格隔开。

输出：

输出vector中的所有整数，每行一个。

样例：

3
1 2 3

3
2
1

#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
void Insert(vector<int> &qy,int x)
{
	vector<int>::iterator it;
	it=find(qy.begin(),qy.end(),x);  //找一遍是否存在x
	if(it==qy.end())  //不存在，则在begin处插入x
	    qy.insert(qy.begin(),x);
}

5364: C++实验：STL之vector3

描述：

将n个数字输入到vector里，并对其进行从大到小排序并输出。
部分代码已经给出，请补充完整，提交时请勿包含已经给出的代码。

 int main()
{
    vector<int> vec;
    while(Input(vec))
    {
        Sort(vec);
        for(vector<int>::iterator it=vec.begin();it!=vec.end();it++)
        {
            if(it!=vec.begin())
                cout<<" ";
            cout<<*it;
        }
        cout<<endl;
    }

    return 0;
}

输入：

输入数据有多组，每组第一行为正整数n，第二行为n个整数，空格隔开。
输入以EOF结束。

输出：

输出从大到小排序后的值。

样例：

3
1 3 2
4
1 2 3 4

3 2 1
4 3 2 1

#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
bool cmp(int a,int b)
{
	return a>b;
}
void Sort(vector<int> &qy)
{
	sort(qy.begin(),qy.end(),cmp);
}
int Input(vector<int> &qy)
{
	qy.clear();
	int n,x;
	while(cin>>n)
	{
		while(n--)
		{
			cin>>x;
			qy.push_back(x);
		}
		return 1;
	}
	return 0;
}

5365: C++实验：STL之vector4

描述：

将一个n行m列矩阵元素输入存储后并输出。
部分代码已经给出，请补充完整，提交时请勿包含已经给出的代码。

int main()
{
    vector< vector<int> > vec;
    Input(vec);
    for(int i=0;i<vec.size();i++)
    {
        for(int j=0;j<vec[i].size();j++)
        {
            if(j)
                cout<<" ";
            cout<<vec[i][j];
        }
        cout<<endl;
    }
    return 0;
}

输入：

第一行为两个整数n和m，接下来有n行，每行m个整数。

输出：

输出n行，每行m个整数，空格隔开。

样例：

2 3
1 2 3
3 4 5

1 2 3
3 4 5

#include<iostream>
#include<vector>
using namespace std;
void Input(vector< vector<int> > &qy)
{
	vector<int> kk; 
	int n,m,x;
	cin>>n>>m;
	for(int i=0;i<n;i++)
	{
		kk.clear();
		for(int j=0;j<m;j++)
		{
			cin>>x;
			kk.push_back(x);
		}
		qy.push_back(kk);
	}
}
