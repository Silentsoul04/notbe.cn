[剑指offer]7、斐波那契数列（三种方法）

May 11, 2019

题目描述

大家都知道斐波那契数列，现在要求输入一个整数n，请你输出斐波那契数列的第n项（从0开始，第0项为0）。

n<=39

法一：可以容易得到推导公式：，递归可求，时间复杂度为，不推荐此法。

法二：根据公式，可以用F1记录F(n-2)，F2记录F(n-1)，依次相加，并更新F1和F2的值即可，时间复杂度为，代码如下：

int Fibonacci(int n) {
        if(n < 2)
            return n;
        int F1 = 0, F2 = 1, res = 0;
        for(int i = 2; i <= n; i++)
        {
            res = F1 + F2;
            F1 = F2;
            F2 = res;
        }
        return res;
    }

法三：推导公式可以通过矩阵的乘法表示，，根据F(1) = 1, F(2) = 1, F(3) = 2, F(4) = 3，可求出状态矩阵。根据推导，可得，则求出二阶矩阵就可求出F(n)。二阶矩阵的n次幂可借鉴整数的n次幂求解方法https://blog.csdn.net/weixin_41814716/article/details/90114636，将整数换成二阶矩阵求解即可，其中二阶矩阵相乘公式。该方法的时间复杂度为，代码如下：

 int Fibonacci(int n) {
        vector<vector<int> >A(2, vector<int>(2, 1));
	    A[1][1] = 0;
	    if (n < 2)
		    return n;
	    vector<vector<int> >res = PowMatrix(A, n - 2);
	    return res[0][0] + res[1][0];
    }
//矩阵相乘
    vector<vector<int> >MultiMatrix(vector<vector<int> >&a, vector<vector<int> >&b)
{
	int row = a.size();
	int col = b[0].size();
	vector<vector<int> >res(row, vector<int>(col, 0));
	for (int i = 0; i < row; i++)
	{
		for (int j = 0; j < col; j++)
		{
			for (int k = 0; k < b.size(); k++)
			{
				res[i][j] += a[i][k] * b[k][j];
			}
		}
	}
	return res;
}
//求矩阵的n次幂
vector<vector<int> >PowMatrix(vector<vector<int> >&arr, int n)
{
	int row = arr.size();
	int col = arr[0].size();
	vector<vector<int> >res(row, vector<int>(col, 0));
	for (int i = 0; i < row; i++)
		res[i][i] = 1;
	vector<vector<int> >tmp = arr;
	while (n != 0)
	{
		if (n & 1)
			res = MultiMatrix(res, tmp);
		tmp = MultiMatrix(tmp, tmp);
		n >>= 1;
	}
	return res;
}
