Unity实战篇：读取Excel数据并转换成Asset

2018年10月17日 20:15:01 烟雨迷离半世殇 阅读数：332更多

所属专栏： Unity实战汇总

 版权声明：本文为博主原创文章，未经博主允许不得转载。 https://blog.csdn.net/qq_15020543/article/details/83098127

在开发游戏过程中我们不可避免的会遇到要给游戏数据配表的问题，（毕竟一个一个string写太累了啊喂！），而且配表还有利于数据的观察和策略性修改，也有利于游戏的维护与更新。这篇博客讲的是读取Excel表格数据，并且转化为Asset文件。

它具有以下优点

  • 我们可以不用将Excel文件放到工程里，只需要生成一次Asset文件即可删除
  • 解决了项目打包成EXE文件无法读取Excel表格的问题
  • 解决了项目打包成安卓平台无法读取Excel表格的问题
  • 不需要考虑跨平台问题

以下内容部分参考自这位大神博客，我在其基础上加以细节说明。（大神们总是这样，不够体谅我这种菜鸡的感受。）

先放一下表结构图和生成的Asset图

然后开始实现

1.下载读取Excel文件必要的dll文件。

https://download.csdn.net/download/qq_15020543/10725145

把下载的三个dll文件放到Unity的Plugins文件夹下面，如果没有就创建。

2.创建Excel文件，名称随意。我们这里取名为Demo，注意后缀一定是xlsx，放在Excel文件夹下面

创建代码名为ExcelReader（随意），读取Excel文件，并做编辑器拓展，转化成Asset文件。注意一定要放在Editor文件夹下面！！！

 


 

  1. using System.Collections;

  2. using System.Collections.Generic;

  3. using System.Data;

  4. using Excel;

  5. using System.IO;

  6. using UnityEngine;

  7. using UnityEditor;

  8. using Data;

  9.  
  10.  

  11. public class ExcelReader {

  12.  
  13.  
  14.  

  15. public class ExcelConfig

  16. {

  17. /// <summary>

  18. /// 存放excel表文件夹的的路径，本例xecel表放在了"Assets/Excels/"当中

  19. /// </summary>

  20. public static readonly string excelsFolderPath = Application.dataPath + "/Excels/";

  21.  

  22. /// <summary>

  23. /// 存放Excel转化CS文件的文件夹路径

  24. /// </summary>

  25. public static readonly string assetPath = "Assets/Resources/";

  26. }

  27.  

  28. public class ExcelTool

  29. {

  30.  

  31. /// <summary>

  32. /// 读取表数据，生成对应的数组

  33. /// </summary>

  34. /// <param name="filePath">excel文件全路径</param>

  35. /// <returns>Item数组</returns>

  36. public static Item[] CreateItemArrayWithExcel(string filePath)

  37. {

  38. //获得表数据

  39. int columnNum = 0, rowNum = 0;

  40. DataRowCollection collect = ReadExcel(filePath, ref columnNum, ref rowNum);

  41.  

  42. //根据excel的定义，第二行开始才是数据

  43. Item[] array = new Item[rowNum - 1];

  44. for (int i = 1; i < rowNum; i++)

  45. {

  46. Item item = new Item();

  47. //解析每列的数据

  48. item.itemId = collect[i][0].ToString();

  49. item.itemLink = collect[i][1].ToString();

  50. item.itemDamage = collect[i][2].ToString();

  51. array[i - 1] = item;

  52. }

  53. return array;

  54. }

  55.  

  56. /// <summary>

  57. /// 读取excel文件内容

  58. /// </summary>

  59. /// <param name="filePath">文件路径</param>

  60. /// <param name="columnNum">行数</param>

  61. /// <param name="rowNum">列数</param>

  62. /// <returns></returns>

  63. static DataRowCollection ReadExcel(string filePath, ref int columnNum, ref int rowNum)

  64. {

  65. FileStream stream = File.Open(filePath, FileMode.Open, FileAccess.Read, FileShare.Read);

  66. IExcelDataReader excelReader = ExcelReaderFactory.CreateOpenXmlReader(stream);

  67.  

  68. DataSet result = excelReader.AsDataSet();

  69. //Tables[0] 下标0表示excel文件中第一张表的数据

  70. columnNum = result.Tables[0].Columns.Count;

  71. rowNum = result.Tables[0].Rows.Count;

  72. return result.Tables[0].Rows;

  73. }

  74. }

  75.  

  76. public class ExcelBuild : Editor

  77. {

  78.  

  79. [MenuItem("CustomEditor/CreateItemAsset")]

  80. public static void CreateItemAsset()

  81. {

  82. ItemManager manager = ScriptableObject.CreateInstance<ItemManager>();

  83. //赋值

  84. manager.dataArray = ExcelTool.CreateItemArrayWithExcel(ExcelConfig.excelsFolderPath + "Demo.xlsx");

  85.  

  86. //确保文件夹存在

  87. if (!Directory.Exists(ExcelConfig.assetPath))

  88. {

  89. Directory.CreateDirectory(ExcelConfig.assetPath);

  90. }

  91.  

  92. //asset文件的路径 要以"Assets/..."开始，否则CreateAsset会报错

  93. string assetPath = string.Format("{0}{1}.asset", ExcelConfig.assetPath, "Item");

  94. //生成一个Asset文件

  95. AssetDatabase.CreateAsset(manager, assetPath);

  96. AssetDatabase.SaveAssets();

  97. AssetDatabase.Refresh();

  98. }

  99. }

  100.  
  101.  

  102. }
