python的三级菜单

May 11, 2019

  1. python的三级菜单用的最多的是循环，来个笨方法看看
data = {
    "陕西":{
        "汉中":{
            "城固":["城固二中","城固一中"],
            "洋县":["洋县一中","洋县二中"]
        },
        "西安":{
            "长安区":["长安一中","长安二中"],
            "雁塔区":["雁塔一中","雁塔二中"],
            "未央区":[],
            "灞桥区":[],
        },
        "咸阳":{},
        "榆林":{}
    },
    "重庆":{
    },
    "四川":{}
}
exit_flag = False

while not exit_flag:
    for i in data:
        print(i)        #打印第一层
    choice = input("选择进入1>>>：")
    if choice in data:
        while not exit_flag:
            for i2 in data[choice]:
                print("\t",i2)      #打印第二层
            choice2 = input("选择进入2>>>:")
            if choice2 in data[choice]:
                while not exit_flag:
                    for i3 in data[choice][choice2]:
                        print("\t\t",i3)        #打印第三层
                    choice3 = input("选择进入3>>>:")
                    if choice3 in data[choice][choice2]:        #输入不一个存在的
                        for i4 in data[choice][choice2][choice3]:
                            print("\t\t",i4)
                        choice4 = input("最后一层，按q返回>>:")
                        if choice4 == "b":
                            pass
                        elif choice4 == "q":
                            exit_flag = True
                    if choice3 == "b":
                        break
                    elif choice3 == "q":
                        exit_flag = True
            if choice2 == "b":
                break
            elif choice2 == "q":
                exit_flag = True
