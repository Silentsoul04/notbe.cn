编写一个完整的包含输入和输出的简单程序。 （1）密码错误，显示“对不起，密码错误！”。 （2）密码正确，显示“欢迎你！”，同时内置函数求解2x2+4x+5的值，x为整数，并用主函数调用该函数。 （3）

May 11, 2019

#include<iostream>

using namespace std;

inline int f(int x)

{

    return 2*x*x+4*x+5;

}

inline double f(double x)

{

    return 2*x*x+4*x+5;

}

int main()

{

    cout<<"密码保护程序"<<endl;

    int pass;

    cout<<"请输入口令：";

    cin>>pass;

    if(pass==12345)

    {

        cout<<"欢迎你！"<<endl;

        int x=5;

        double y=5.2;

        cout<<f(x)<<endl;

        cout<<f(y)<<endl;

    }

 

    else

        cout<<"对不起，密码错误！"<<endl;

    return 0;

}
