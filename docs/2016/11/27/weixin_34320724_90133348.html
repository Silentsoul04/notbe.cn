然后再对数据，进行字符串处理。

foreach ($researchAwardCat as $i => $ival) {//论文
            $outData[$i-1]['name'] = $ival;
            $outstr = '';
            $count = 0;
            foreach ($researchAwardLevel as $j => $jval) {//国际、国家
                if($count!=0){
                    $outstr.=',';
                }
                if(isset($a[$i][$j])){
                        $outstr .=$a[$i][$j]['count'];
                }else{
                        $outstr .='0';
                }
                $count++;
            }
            $outData[$i-1]['data'] = $outstr;
        }
