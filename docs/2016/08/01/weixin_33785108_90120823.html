        public static int count(int n)
        {
            int max = 1592;
            int min = 0;
            int cur=0;
            while (max >= min)
            {
                cur = (max + min) / 2;
                if (n < arr[cur])
                {
                    max = cur - 1;
                }
                else if (n > arr[cur])
                {
                    min = cur + 1;
                }
                else
                {
                    break;
                }
            }
            if (max < min)
            {
                cur = max;
            }
            return cur+1;
        }
