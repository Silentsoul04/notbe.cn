  4.         private void webBrowser1_PreviewKeyDown(object sender, PreviewKeyDownEventArgs e)
  5.         {
  6.             if (e.Modifiers == Keys.Control && e.KeyCode == Keys.S)
  7.             {
  8.                 if (this.Name.Equals("editorChild") && this.ParentForm is EditExamQuestion)
  9.                     ((EditExamQuestion)this.ParentForm).ChildNodeSave();

  11.                 if (this.Name.Equals("editorRoot") && this.ParentForm is EditExamQuestion)
  12.                     ((EditExamQuestion)this.ParentForm).RootNodeSave();

  14.                 if (this.Name.Equals("editorTopic") && this.ParentForm is EditExamQuestion)
  15.                     ((EditExamQuestion)this.ParentForm).EditorTopicSave();
  16.             }
  17.             
  18.             if (e.KeyData == Keys.Tab)
  19.                 Indent();

  21.             if (e.Modifiers == Keys.Shift && e.KeyCode == Keys.Tab)
  22.                 Outdent();

  24.             this.webBrowser1.Focus();
  25.         }
