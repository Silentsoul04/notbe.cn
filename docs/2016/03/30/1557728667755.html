C# Http下载指定文件存储到特定目录

Mar 30, 2016

    C# Http下载指定文件存储到特定目录，下面是标准用法代码：


  1. ///
  2. /// Http下载文件：下载指定文件存储到特定目录
  3. ///
  4. public static string HttpDownloadFile(string remoteUrl)
  5. {
  6.     // 设置参数
  7.     HttpWebRequest request = WebRequest.Create(remoteUrl) as HttpWebRequest;
  8.     //发送请求并获取相应回应数据
  9.     HttpWebResponse response = request.GetResponse() as HttpWebResponse;
  10.     //直到request.GetResponse()程序才开始向目标网页发送Post请求
  11.     Stream responseStream = response.GetResponseStream();

  13.     string localPath = AppDomain.CurrentDomain.BaseDirectory + @"QRCode\MyQRCode.jpg";
  14.     //创建本地文件写入流
  15.     Stream stream = new FileStream(localPath, FileMode.Create);
  16.     byte[] bArr = new byte[1024];
  17.     int size = responseStream.Read(bArr, 0, (int)bArr.Length);
  18.     while (size > 0)
  19.     {
  20.         stream.Write(bArr, 0, size);
  21.         size = responseStream.Read(bArr, 0, (int)bArr.Length);
  22.     }
  23.     stream.Close();
  24.     responseStream.Close();
  25.     return localPath;
  26. }
