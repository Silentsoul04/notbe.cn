[user]
         name = Florent Lebreton
         email = florentlebreton@ free .fr
[color]
         ui = auto
 
[color  "branch" ]
         current = yellow reverse
         local  = yellow
         remote = green
 
[color  "diff" ]
         meta = yellow bold
         frag = magenta bold
         old = red bold
         new = green bold
         whitespace = red reverse
 
[color  "status" ]
         added = yellow
         changed = green
         untracked = cyan
 
[core]
         whitespace=fix,-indent-with-non-tab,trailing-space,cr-at-eol
 
[ alias ]
         st = status
         ci = commit
         br = branch
         co = checkout
         df  =  diff
         dc  =  diff  --cached
         lg = log -p
         pr  = pull --rebase
         gr = log --all --graph --decorate --oneline
