 def  cfsize(cha):

     else  : 
        fsize = 0
        fnum = cfsize( ' Z ' ) + 1
        cc = list(cha)
         for  tnum  in  range(len(cc)):
            fsize += fnum * tnum + cfsize(cc[tnum])
