' VBScript\
Dim message
Sub myPrint(msg)
   Const ForReading = 1, ForWriting = 2, ForAppending = 8
   Dim fso, f
   Set fso = CreateObject("Scripting.FileSystemObject")
   Set f = fso.OpenTextFile("D:\QTPscript.log", ForAppending, True)
   f.Write "测试执行结果步骤: "&"时间："&now()&vbnewline&"测试信息："&msg&vbnewline
   f.Close
End Sub
