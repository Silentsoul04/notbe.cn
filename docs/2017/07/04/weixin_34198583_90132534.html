PUBLIC "-//ibatis.apache.org//DTD Config 3.0//EN"
"http://ibatis.apache.org/dtd/ibatis-3-config.dtd">
<configuration>
<typeAliases>
<typeAlias alias="myuser" type="com.action.Users" />
</typeAliases>
<!-- 加载jdbc.properties -->
    <properties resource="jdbc.properties"/> 
    <environments default="development"> 
        <environment id="development">
            <transactionManager type="JDBC"/> 
            <dataSource type="POOLED">
                <property name="driver" value="driver"/><propertyname="url"value="{url}"/>
                <property name="username" value="username"/><propertyname="password"value="{password}"/> 
            </dataSource> 
        </environment> 
    </environments>
    <!-- 加载映射文件 -->
    <mappers>
        <mapper resource="com/action/User.xml"/> 
    </mappers>
</configuration>
