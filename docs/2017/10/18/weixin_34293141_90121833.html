1、新建CheckCodeImageExt.aspx

using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;
using System.Drawing;

namespace CheckCode
{
    public partial class CheckCodeImageExt : System.Web.UI.Page
    {
        protected void Page_Load(object sender, EventArgs e)
        {
            CreateCheckCodeImage(GenarateCheckCode());
        }
        private string GenarateCheckCode()  
        {
            int number;
            char code;
            string CheckCode = string.Empty;

            System.Random random = new Random();
            for (int i = 0; i < 5; i++)
            {
                number = random.Next();
                if (number % 2 == 0)
                    code = (char)('0' + (char)(number % 10));
                else
                    code = (char)('A' + (char)(number % 26));
                CheckCode += code.ToString();
            }
            Session["CheckCode"] = CheckCode;
            return CheckCode;
        }

        private void CreateCheckCodeImage(string checkCode)
        {
            if (checkCode == null || checkCode.Trim() == string.Empty)
                return;

            System.Drawing.Bitmap image = new System.Drawing.Bitmap((int)Math.Ceiling(checkCode.Length * 12.5), 24);//宽度，高度
            Graphics g = Graphics.FromImage(image);

            try
            {
                Random random = new Random();
                //清空图片背景色，并用白色填充
                g.Clear(Color.White);
                //画图片的背景噪音线
                for (int i = 0; i < 25; i++)
                {
                    int x1 = random.Next(image.Width);
                    int x2 = random.Next(image.Width);
                    int y1 = random.Next(image.Height);
                    int y2 = random.Next(image.Height);

                    g.DrawLine(new Pen(Color.Silver), x1, y1, x2, y2);
                }

                Font font = new System.Drawing.Font("Arial", 12, (System.Drawing.FontStyle.Bold | System.Drawing.FontStyle.Italic));
                System.Drawing.Drawing2D.LinearGradientBrush brush = new System.Drawing.Drawing2D.LinearGradientBrush(new Rectangle(0, 0, image.Width, image.Height), Color.Blue, Color.DarkRed, 1.2f, true);
                g.DrawString(checkCode, font, brush, 2, 2);


                //画图片的前景噪音点
                for (int i = 0; i < 100; i++)
                {
                    int x = random.Next(image.Width);
                    int y = random.Next(image.Height);

                    image.SetPixel(x, y, Color.FromArgb(random.Next()));
                }
                //画图片的边框线
                g.DrawRectangle(new Pen(Color.Silver), 0, 0, image.Width - 1, image.Height - 1);
                System.IO.MemoryStream ms = new System.IO.MemoryStream();
                image.Save(ms, System.Drawing.Imaging.ImageFormat.Gif);
                Response.ClearContent();
                Response.ContentType = "image/Gif";
                Response.BinaryWrite(ms.ToArray());
            }
            catch
            {
                g.Dispose();
                image.Dispose();
            }
        }
    }
}

    <title></title>
    <script type="text/javascript">
        function reload() {
            var code = document.getElementById('imgCode');
            code.src += '?';
        }
    </script>
</head>
<body>
    <form id="form1" runat="server">
    <div>
        <div style=" height:200px;"></div>
        <div style="float:left; width:30%; height:100px;"></div>
        <div style="float:left; width:40%; height:100px;">
            <div><img src="CheckCodeImageExt.aspx" alt="checkCode" id="imgCode" onclick="reload();" title="点击刷新" /></div>
        </div>
    </div>
    </form>
</body>
</html>

 

img.src = img.src+'?';这一句使服务器重新生成图片, 问号的作用是忽悠浏览器, 不要让浏览器缓存
