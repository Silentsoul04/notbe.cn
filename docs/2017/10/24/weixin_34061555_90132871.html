Xamarin.Forms快速入门

Oct 24, 2017

本演练演示如何创建将用户输入的字母数字电话号码转换为数字电话号码的应用程序，并调用该号码。 最终的应用如下所示：

创建Phoneword应用程序如下：

  1. 在“开始”屏幕中，启动Visual Studio。 这将打开起始页：

  2. 在Visual Studio中，单击创建新项目...以创建新项目：

  3. 在“新建项目”对话框中，单击“跨平台”，选择“跨平台应用程序（Xamarin.Forms或Native）”模板，将名称和解决方案名称设置为Phoneword，为项目选择合适的位置，然后单击确定按钮：

  4. 在“新跨平台应用程序”对话框中，单击空白应用程序，选择Xamarin.Forms作为UI技术，选择便携式类库（PCL）作为代码共享策略，然后单击确定按钮：

  5. 在新的通用Windows项目对话框中，选择通用Windows平台（UWP）应用程序将支持的目标和最低版本的Windows 10：

  6. 在解决方案资源管理器中，右键单击Phoneword解决方案，然后选择Manage NuGet Packages for Solution ...：

  7. 在“管理软件包解决方案”对话框中，选择更新选项卡，选择Xamarin.Forms软件包，并将软件包更新为最新的稳定版本：

  8. 在解决方案资源管理器中，在Phoneword项目中，双击MainPage.xaml打开它：

  9. 在MainPage.xaml中，删除所有的模板代码，并将其替换为以下代码。 该代码声明性地定义了页面的用户界面：

    通过按CTRL+S将更改保存到MainPage.xaml，然后关闭文件。

    4. namespace Phoneword

    13.         }

    15.         void OnTranslate (object sender, EventArgs e)

    17.             translatedNumber = Core.PhonewordTranslator.ToNumber (phoneNumberText.Text);
    18.             if (!string.IsNullOrWhiteSpace (translatedNumber)) {
    19.                 callButton.IsEnabled = true;
    20.                 callButton.Text = "Call " + translatedNumber;
    21.             } else {
    22.                 callButton.IsEnabled = false;
    23.                 callButton.Text = "Call";

    27.         async void OnCall (object sender, EventArgs e)
    28.         {

    30.                     "Dial a Number",
    31.                     "Would you like to call " + translatedNumber + "?",
    32.                     "Yes",
    33.                     "No")) {
    34.                 var dialer = DependencyService.GetIDialer> ();
    35.                 if (dialer != null)
    36.                     dialer.Dial (translatedNumber);
    37.             }
    38.         }
    39.     }
    40. }

  14. 在解决方案资源管理器中，右键单击Phoneword项目，然后选择添加>新建项目...：

  15. 在添加新项目对话框中，选择Visual C＃>代码>类，命名新文件PhoneTranslator，然后单击添加按钮：

  16. 在PhoneTranslator.cs中，删除所有的模板代码，并将其替换为以下代码。 这段代码将把手机字词翻译成电话号码：

    点击(此处)折叠或打开

    1. using System.Text;

    3. namespace Core
    4. {
    5.     public static class PhonewordTranslator
    6.     {
    7.         public static string ToNumber(string raw)
    8.         {
    9.             if (string.IsNullOrWhiteSpace(raw))
    10.                 return null;

    12.             raw = raw.ToUpperInvariant();

    14.             var newNumber = new StringBuilder();
    15.             foreach (var c in raw)
    16.             {
    17.                 if (" -0123456789".Contains(c))
    18.                     newNumber.Append(c);
    19.                 else
    20.                 {
    21.                     var result = TranslateToNumber(c);
    22.                     if (result != null)
    23.                         newNumber.Append(result);
    24.                     // Bad character?
    25.                     else
    26.                         return null;
    27.                 }
    28.             }
    29.             return newNumber.ToString();
    30.         }

    32.         static bool Contains(this string keyString, char c)
    33.         {
    34.             return keyString.IndexOf(c) >= 0;
    35.         }

    37.         static readonly string[] digits = {
    38.             "ABC", "DEF", "GHI", "JKL", "MNO", "PQRS", "TUV", "WXYZ"
    39.         };

    41.         static int? TranslateToNumber(char c)
    42.         {
    43.             for (int i = 0; i digits.Length; i++)
    44.             {
    45.                 if (digits[i].Contains(c))
    46.                     return 2 + i;
    47.             }
    48.             return null;
    49.         }
    50.     }
    51. }

    1. using Android.Content;
    2. using Android.Telephony;
    3. using Phoneword.Droid;
    4. using System.Linq;
    5. using Xamarin.Forms;
    6. using Uri = Android.Net.Uri;

    8. [assembly: Dependency(typeof(PhoneDialer))]
    9. namespace Phoneword.Droid
    10. {
    11.     public class PhoneDialer : IDialer
    12.     {
    13.         public bool Dial(string number)
    14.         {
    15.             var context = Forms.Context;
    16.             if (context == null)
    17.                 return false;

    19.             var intent = new Intent (Intent.ActionCall);
    20.             intent.SetData (Uri.Parse ("tel:" + number));

    22.             if (IsIntentAvailable (context, intent)) {
    23.                 context.StartActivity (intent);
    24.                 return true;
    25.             }

    27.             return false;
    28.         }

    30.         public static bool IsIntentAvailable(Context context, Intent intent)
    31.         {
    32.             var packageManager = context.PackageManager;

    34.             var list = packageManager.QueryIntentServices (intent, 0)
    35.                 .Union (packageManager.QueryIntentActivities (intent, 0));

    37.             if (list.Any ())
    38.                 return true;

    40.             var manager = TelephonyManager.FromContext (context);
    41.             return manager.PhoneType != PhoneType.None;
    42.         }
    43.     }
    44. }

    1. using Phoneword.UWP;
    2. using System;
    3. using System.Threading.Tasks;
    4. using Windows.ApplicationModel.Calls;
    5. using Windows.UI.Popups;
    6. using Xamarin.Forms;

    8. [assembly: Dependency(typeof(PhoneDialer))]
    9. namespace Phoneword.UWP
    10. {
    11.     public class PhoneDialer : IDialer
    12.     {
    13.         bool dialled = false;

    15.         public bool Dial(string number)
    16.         {
    17.             DialNumber(number);
    18.             return dialled;
    19.         }

    21.         async Task DialNumber(string number)
    22.         {
    23.             var phoneLine = await GetDefaultPhoneLineAsync();
    24.             if (phoneLine != null)
    25.             {
    26.                 phoneLine.Dial(number, number);
    27.                 dialled = true;
    28.             }
    29.             else
    30.             {
    31.                 var dialog = new MessageDialog("No line found to place the call");
    32.                 await dialog.ShowAsync();
    33.                 dialled = false;
    34.             }
    35.         }

    37.         async TaskPhoneLine> GetDefaultPhoneLineAsync()
    38.         {
    39.             var phoneCallStore = await PhoneCallManager.RequestStoreAsync();
    40.             var lineId = await phoneCallStore.GetDefaultLineAsync();
    41.             return await PhoneLine.FromIdAsync(lineId);
    42.         }
    43.     }
    44. }
