  19. package org.apache.tools.ant.listener;

  21. import java.util.Map;
  22. import org.apache.logging.log4j.LogManager;
  23. import org.apache.logging.log4j.Logger;
  24. import org.apache.logging.log4j.core.Appender;
  25. import org.apache.tools.ant.BuildEvent;
  26. import org.apache.tools.ant.BuildListener;
  27. import org.apache.tools.ant.Project;
  28. import org.apache.tools.ant.Target;
  29. import org.apache.tools.ant.Task;

  31. /**
  32. * Listener which sends events to Log4j logging system
  33. *
  34. */
  35. public class Log4j2Listener implements BuildListener {

  37.     /** Indicates if the listener was initialized. */
  38.     private final boolean initialized;

  40.     /**
  41.      * log category we log into
  42.      */
  43.     public static final String LOG_ANT = "org.apache.tools.ant";

  45.     /**
  46.      * Construct the listener and make sure there is a valid appender.
  47.      */
  48.     public Log4j2Listener() {
  49.         final Logger log = LogManager.getLogger(LOG_ANT);
  50.         final Logger rootLog = LogManager.getRootLogger();
  51.         MapString, Appender> appenderMap =
  52.                 ((org.apache.logging.log4j.core.Logger) rootLog).getAppenders();
  53.         initialized = !(appenderMap.isEmpty());
  54.         if (!initialized) {
  55.             log.error("No log4j.properties in build area");
  56.         }
  57.     }

  59.     /**
  60.      * @see BuildListener#buildStarted
  61.      */
  62.     /** {@inheritDoc}. */
  63.     public void buildStarted(final BuildEvent event) {
  64.         if (initialized) {
  65.             final Logger log = LogManager.getLogger(Project.class.getName());
  66.             log.info("Build started.");
  67.         }
  68.     }

  70.     /**
  71.      * @see BuildListener#buildFinished
  72.      */
  73.     /** {@inheritDoc}. */
  74.     public void buildFinished(final BuildEvent event) {
  75.         if (initialized) {
  76.             final Logger log = LogManager.getLogger(Project.class.getName());
  77.             if (event.getException() == null) {
  78.                 log.info("Build finished.");
  79.             } else {
  80.                 log.error("Build finished with error.", event.getException());
  81.             }
  82.         }
  83.     }

  85.     /**
  86.      * @see BuildListener#targetStarted
  87.      */
  88.     /** {@inheritDoc}. */
  89.     public void targetStarted(final BuildEvent event) {
  90.         if (initialized) {
  91.             final Logger log = LogManager.getLogger(Target.class.getName());
  92.             log.info("Target \"" + event.getTarget().getName() + "\" started.");
  93.         }
  94.     }

  96.     /**
  97.      * @see BuildListener#targetFinished
  98.      */
  99.     /** {@inheritDoc}. */
  100.     public void targetFinished(final BuildEvent event) {
  101.         if (initialized) {
  102.             final String targetName = event.getTarget().getName();
  103.             final Logger cat = LogManager.getLogger(Target.class.getName());
  104.             if (event.getException() == null) {
  105.                 cat.info("Target \"" + targetName + "\" finished.");
  106.             } else {
  107.                 cat.error("Target \"" + targetName
  108.                     + "\" finished with error.", event.getException());
  109.             }
  110.         }
  111.     }

  113.     /**
  114.      * @see BuildListener#taskStarted
  115.      */
  116.     /** {@inheritDoc}. */
  117.     public void taskStarted(final BuildEvent event) {
  118.         if (initialized) {
  119.             final Task task = event.getTask();
  120.             final Logger log = LogManager.getLogger(task.getClass().getName());
  121.             log.info("Task \"" + task.getTaskName() + "\" started.");
  122.         }
  123.     }

  125.     /**
  126.      * @see BuildListener#taskFinished
  127.      */
  128.     /** {@inheritDoc}. */
  129.     public void taskFinished(final BuildEvent event) {
  130.         if (initialized) {
  131.             final Task task = event.getTask();
  132.             final Logger log = LogManager.getLogger(task.getClass().getName());
  133.             if (event.getException() == null) {
  134.                 log.info("Task \"" + task.getTaskName() + "\" finished.");
  135.             } else {
  136.                 log.error("Task \"" + task.getTaskName()
  137.                     + "\" finished with error.", event.getException());
  138.             }
  139.         }
  140.     }

  142.     /**
  143.      * @see BuildListener#messageLogged
  144.      */
  145.     /** {@inheritDoc}. */
  146.     public void messageLogged(final BuildEvent event) {
  147.         if (initialized) {
  148.             Object categoryObject = event.getTask();
  149.             if (categoryObject == null) {
  150.                 categoryObject = event.getTarget();
  151.                 if (categoryObject == null) {
  152.                     categoryObject = event.getProject();
  153.                 }
  154.             }

  156.             final Logger log
  157.                 = LogManager.getLogger(categoryObject.getClass().getName());
  158.             switch (event.getPriority()) {
  159.                 case Project.MSG_ERR:
  160.                     log.error(event.getMessage());
  161.                     break;
  162.                 case Project.MSG_WARN:
  163.                     log.warn(event.getMessage());
  164.                     break;
  165.                 case Project.MSG_INFO:
  166.                     log.info(event.getMessage());
  167.                     break;
  168.                 case Project.MSG_VERBOSE:
  169.                     log.debug(event.getMessage());
  170.                     break;
  171.                 case Project.MSG_DEBUG:
  172.                     log.debug(event.getMessage());
  173.                     break;
  174.                 default:
  175.                     log.error(event.getMessage());
  176.                     break;
  177.             }
  178.         }
  179.     }
  180. }
