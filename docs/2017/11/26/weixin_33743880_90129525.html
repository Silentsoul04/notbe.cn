{
    X509Certificate2 certificate =  X509Certificate2(ConfigurationSettings.AppSettings[], ConfigurationSettings.AppSettings[]);

    byte[] signature = StrToByte(Plain);
    RSACryptoServiceProvider provider =  RSACryptoServiceProvider();
    provider.FromXmlString(certificate.PublicKey.Key.ToXmlString());
     provider.VerifyData(bytes, , signature);
