int  f[9]={0,10,100,1000,10000,100000,1000000,10000000,100000000};
int  main()
{
     int  a,b,n;
     while  ( scanf ( "%d%d%d" ,&a,&b,&n),a||b)
     {
         if (a%f[n]==b%f[n])
             printf ( "-1\n" );
         else
             printf ( "%d\n" ,a+b);
     }
}
