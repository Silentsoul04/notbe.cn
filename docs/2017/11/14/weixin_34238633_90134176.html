Spark Core & RDD本质上是离线运算

val conf=new SparkConf().setAppName("myWordCount")
            val sc=new SparkContext(conf)
            //读取数据
            val lines=sc.textFile(input)
            //进行相关计算
            val resultRdd=lines
                .flatMap(_.split(" "))
                .map((_,1))
                .reduceByKey(_+_)
            //保存结果
            resultRdd.saveAsTextFile(output)
