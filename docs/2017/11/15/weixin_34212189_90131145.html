真言

题目

百练 2818

思路

1 暴力法，计算一次。移动一次

2优化： 计算一次。跟踪一次。再计算，在跟踪；最后在移动

3再优化：计算须要移动的次数，计算终于结果。然后移动

bool my_2818(vector<string> * myvector)
{
	int n;
	cin>>n;
	if(n == 0)
	{
		return false;
	}
	int * data = new int[n];
	int *length = new int[n];
	for(int i=0;i<n;i++)
	{
		cin>>data[i];
		length[i] = 0;
	}

	for(int j,i=0;i<n;i++)
	{		
		j = data[i]-1;
		length[i]++;
		while(true)
		{
			if(j == i)
				break;
			j = data[j]-1;
			length[i]++;
		}
	}

	int k;
	string a;
	char* b = new char[n];
	char c;
	while(true)
	{
		cin>>k;
		if(k == 0)
			break;
		for(int i=0;i<n;i++)
		{
			b[i] = ' ';
		}
		getline(cin,a,'\n');
		a=a.substr(1,a.length()-1);
		while(a.length()<n)
			a+=' ';

		for(int i=0;i<n;i++)
		{
			int j = k % length[i];
			int m = i;
			while(j>0)
			{
				m = data[m]-1;
				j--;
			}
			b[m] = a[i];
		}
		string result="";
		for(int j=0;j < n;j++)
			result+=b[j];

		myvector->push_back(result);
	}
	myvector->push_back("");
	return true;
}
