XML与DataSet的相互转换

Nov 3, 2017

XmlDatasetConvert 该类提供了四种方法： 

    1、将xml对象内容字符串转换为DataSet 
    2、将xml文件转换为DataSet 
    3、将DataSet转换为xml对象字符串 
    4、将DataSet转换为xml文件


XmlDatasetConvert.cs 

  1. using System;
  2. using System.Collections.Generic;
  3. using System.Text;
  4. using System.Data;
  5. using System.IO;
  6. using System.Xml;
  8. namespace XmlDesign
  9. {
  10.     class XmlDatasetConvert
  11.     {
  12.         //将xml对象内容字符串转换为DataSet
  13.         public static DataSet ConvertXMLToDataSet(string xmlData)
  14.         {
  15.             StringReader stream = null;
  16.             XmlTextReader reader = null;
  17.             try
  18.             {
  19.                 DataSet xmlDS = new DataSet();
  20.                 stream = new StringReader(xmlData);
  21.                 //从stream装载到XmlTextReader
  22.                 reader = new XmlTextReader(stream);
  23.                 xmlDS.ReadXml(reader);
  24.                 return xmlDS;
  25.             }
  26.             catch (System.Exception ex)
  27.             {
  28.                 throw ex;
  29.             }
  30.             finally
  31.             {
  32.                 if (reader != null)
  33.                     reader.Close();
  34.             }
  35.         }
  37.         //将xml文件转换为DataSet
  38.         public static DataSet ConvertXMLFileToDataSet(string xmlFile)
  39.         {
  40.             StringReader stream = null;
  41.             XmlTextReader reader = null;
  42.             try
  43.             {
  44.                 XmlDocument xmld = new XmlDocument();
  45.                 xmld.Load(xmlFile);
  47.                 DataSet xmlDS = new DataSet();
  48.                 stream = new StringReader(xmld.InnerXml);
  49.                 //从stream装载到XmlTextReader
  50.                 reader = new XmlTextReader(stream);
  51.                 xmlDS.ReadXml(reader);
  52.                 //xmlDS.ReadXml(xmlFile);
  53.                 return xmlDS;
  54.             }
  55.             catch (System.Exception ex)
  56.             {
  57.                 throw ex;
  58.             }
  59.             finally
  60.             {
  61.                 if (reader != null)
  62.                     reader.Close();
  63.             }
  64.         }
  66.         //将DataSet转换为xml对象字符串
  67.         public static string ConvertDataSetToXML(DataSet xmlDS)
  68.         {
  69.             MemoryStream stream = null;
  70.             XmlTextWriter writer = null;
  72.             try
  73.             {
  74.                 stream = new MemoryStream();
  75.                 //从stream装载到XmlTextReader
  76.                 writer = new XmlTextWriter(stream, Encoding.Unicode);
  78.                 //用WriteXml方法写入文件.
  79.                 xmlDS.WriteXml(writer);
  80.                 int count = (int) stream.Length;
  81.                 byte[] arr = new byte[count];
  82.                 stream.Seek(0, SeekOrigin.Begin);
  83.                 stream.Read(arr, 0, count);
  85.                 UnicodeEncoding utf = new UnicodeEncoding();
  86.                 return utf.GetString(arr).Trim();
  87.             }
  88.             catch (System.Exception ex)
  89.             {
  90.                 throw ex;
  91.             }
  92.             finally
  93.             {
  94.                 if (writer != null)
  95.                     writer.Close();
  96.             }
  97.         }
  99.         //将DataSet转换为xml文件
  100.         public static void ConvertDataSetToXMLFile(DataSet xmlDS, string xmlFile)
  101.         {
  102.             MemoryStream stream = null;
  103.             XmlTextWriter writer = null;
  105.             try
  106.             {
  107.                 stream = new MemoryStream();
  108.                 //从stream装载到XmlTextReader
  109.                 writer = new XmlTextWriter(stream, Encoding.Unicode);
  111.                 //用WriteXml方法写入文件.
  112.                 xmlDS.WriteXml(writer);
  113.                 int count = (int) stream.Length;
  114.                 byte[] arr = new byte[count];
  115.                 stream.Seek(0, SeekOrigin.Begin);
  116.                 stream.Read(arr, 0, count);
  118.                 //返回Unicode编码的文本
  119.                 UnicodeEncoding utf = new UnicodeEncoding();
  120.                 StreamWriter sw = new StreamWriter(xmlFile);
  121.                 sw.WriteLine("<?xml version=\"1.0\" encoding=\"utf-8\"?>");
  122.                 sw.WriteLine(utf.GetString(arr).Trim());
  123.                 sw.Close();
  124.             }
  125.             catch (System.Exception ex)
  126.             {
  127.                 throw ex;
  128.             }
  129.             finally
  130.             {
  131.                 if (writer != null)
  132.                     writer.Close();
  133.             }
  134.         }
  135.     }
  136. }

  1. using System;
  2. using System.Collections.Generic;
  3. using System.Text;
  4. using System.Xml;
  5. using System.Data;
  7. namespace XmlDesign

  12.         {
  13.             DataSet ds = new DataSet();
  14.             #region 转换一个XML文件(本地\网络均可)为一个DataSet
  15.             //http://news.baidu.com/n?cmd=1&class=sportnews&tn=rss
  16.             //F:\study\001CSharp_Study\002Source\XmlDesign\XmlDesign\Save_Plan.xml

  20.             {
  21.                 PrintTableName(dt.TableName);
